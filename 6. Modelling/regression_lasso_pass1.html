pal <- function(x) {
  f_neg <- scales::col_numeric(
    palette = c(RColorBrewer::brewer.pal(8,"RdBu")[2], '#ffffff'),
    domain = c(max_y[1,]$min, 0)

  )
  ifelse(x < 0, f_neg(x), f_pos(x))
}
